def add_time(start, duration, starting_day=None):
    # Split start time
    start_time, meridian = start.split()
    start_hour, start_minute = map(int, start_time.split(":"))

    # Convert to 24-hour format
    if meridian == "PM":
        start_hour += 12
    if start_hour == 12 and meridian == "AM":
        start_hour = 0

    # Duration split
    dur_hour, dur_minute = map(int, duration.split(":"))

    # Add time
    end_minute = start_minute + dur_minute
    end_hour = start_hour + dur_hour + end_minute // 60
    end_minute = end_minute % 60

    # Calculate days passed
    days_later = end_hour // 24
    end_hour = end_hour % 24

    # Convert back to 12-hour format
    if end_hour == 0:
        display_hour = 12
        meridian = "AM"
    elif end_hour < 12:
        display_hour = end_hour
        meridian = "AM"
    elif end_hour == 12:
        display_hour = 12
        meridian = "PM"
    else:
        display_hour = end_hour - 12
        meridian = "PM"

    # Prepare result
    new_time = f"{display_hour}:{end_minute:02d} {meridian}"

    # Handle day of the week
    if starting_day:
        days = ["Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday", "Sunday"]
        start_index = days.index(starting_day.capitalize())
        new_day = days[(start_index + days_later) % 7]
        new_time += f", {new_day}"

    # Handle day info
    if days_later == 1:
        new_time += " (next day)"
    elif days_later > 1:
        new_time += f" ({days_later} days later)"

    return new_time
